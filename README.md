# agent-dock
agents "dock" into organized slots

agentdock/
├── .vscode/
│   ├── launch.json              # F5 debug config (auto-generated by yo code)
│   ├── tasks.json               # Build tasks
│   └── settings.json            # Workspace settings
│
├── src/
│   ├── extension.ts             # Entry point — activate/deactivate, register everything
│   │
│   ├── models/
│   │   └── session.ts           # Session interface & Category enum
│   │                            #   { id, name, category, status, terminal, createdAt }
│   │
│   ├── managers/
│   │   ├── sessionManager.ts    # Core logic — CRUD for sessions, holds the session list
│   │   │                        #   addSession(), removeSession(), getByCategory(), etc.
│   │   └── terminalManager.ts   # Wraps vscode.Terminal operations
│   │                            #   createTerminal(), focusTerminal(), disposeTerminal()
│   │                            #   listens to onDidCloseTerminal for cleanup
│   │
│   ├── views/
│   │   ├── sessionTreeProvider.ts   # TreeDataProvider — renders the sidebar tree
│   │   │                            #   grouped by category with icons/colors
│   │   └── sessionTreeItem.ts       # TreeItem subclass — single row in the tree
│   │                                #   icon, color, tooltip, context value for menus
│   │
│   ├── commands/
│   │   ├── newSession.ts        # Prompt name + category → create terminal + session
│   │   ├── focusSession.ts      # QuickPick list → terminal.show()
│   │   ├── endSession.ts        # Dispose terminal, remove from manager
│   │   ├── renameSession.ts     # InputBox → update name, refresh tree
│   │   └── changeCategorySession.ts  # QuickPick → update category, refresh tree
│   │
│   ├── constants/
│   │   ├── categories.ts        # Category definitions: label, icon (codicon), color
│   │   │                        #   e.g. { id: 'bug', label: 'Bug Fix', icon: 'bug', color: 'terminal.ansiRed' }
│   │   └── commands.ts          # Command ID strings to avoid typos
│   │                            #   e.g. export const CMD_NEW = 'agentdock.newSession'
│   │
│   └── utils/
│       └── persistence.ts       # Save/load sessions from globalState (Phase 7)
│                                #   For v1: can be a no-op stub
│
├── resources/
│   └── icons/
│       └── agentdock.svg        # Activity bar icon (your logo)
│
├── package.json                 # THE most important file for a VS Code extension:
│   │                            #   contributes.viewsContainers — activity bar icon
│   │                            #   contributes.views — sidebar tree view
│   │                            #   contributes.commands — all 5 commands
│   │                            #   contributes.menus — right-click context menus on tree items
│   │                            #   contributes.keybindings — keyboard shortcuts
│   │                            #   activationEvents — when extension loads
│   │
├── tsconfig.json                # TypeScript config
├── .eslintrc.json               # Linting
├── .gitignore
├── README.md
├── CHANGELOG.md
└── LICENSE
